\begin{algorithm}

\caption{Metamorphic-Mutation-Guided Hallucination Repair via Fault Injection}
\label{alg:mt-repair-ranking}
\KwIn{
    Question $Q$, 
    Mutation Prompt $P_m$, 
    Answer Prompt $P_a$, 
    Ranking Prompt $P_r$, 
    Refinement Prompt $P_f$
}
\KwOut{
    Repaired Answer $A^*$
}

\SetKwFunction{LLM}{LLM}
\SetKwFunction{QType}{QuestionType}
\SetKwProg{Fn}{Function}{:}{}
\SetAlgoNlRelativeSize{0}

\Fn{\textbf{Repair}}{

    \tcp{------------------ Base Response Generation ------------------}
    $B \gets \LLM(Q)$ \tcp{Generate initial (possibly hallucinated) base response}

    \tcp{------------------ Metamorphic Mutation Generation ------------------}
    $M \gets \LLM(Q, B, P_m)$ \tcp{Generate metamorphic variants of base response}
    $M \gets M \cup \{B\}$ \tcp{Include original base response as a mutation variant}

    \tcp{------------------ Answer Regeneration ------------------}
    $A \gets \emptyset$ \tcp{Initialize set of candidate answers}
    \ForEach{$m \in M$}{
        $a_m \gets \LLM(Q, m, P_a)$ \tcp{Re-answer question under mutation}
        $A \gets A \cup \{a_m\}$
    }

    $n \gets |A|$ \tcp{Number of candidate answers (typically $n=6$)}

    \tcp{------------------ Question Type Identification ------------------}
    $\tau \gets \QType(Q)$ \tcp{$\tau \in \{\textsc{Fictional}, \textsc{Subjective}, \textsc{Factual}\}$}

    \tcp{------------------ Pairwise Ground-Truth Ranking ------------------}
    Initialize ranking matrix $R \in \mathbb{N}^{n \times n}$

    \For{$i \gets 1$ \KwTo $n$}{
        \For{$j \gets 1$ \KwTo $n$}{
            $R_{i,j} \gets \LLM(a_i, a_j, Q, \tau, P_r)$ \tcp{
            Compare answer $a_i$ against $a_j$ for the Origin Question Q under judging criteria}            
            Assign rank in $\{1,\dots,n\}$ (lower is better)
            }
        }
    }

    \tcp{------------------ Rank Aggregation ------------------}
    \For{$i \gets 1$ \KwTo $n$}{
        $S(a_i) \gets \sum_{j=1}^{n} R_{i,j}$ \tcp{Compute total ranking score}
    }

    $a^{best} \gets \arg\min_{a_i \in A} S(a_i)$ \tcp{Select lowest-ranked answer}

    \tcp{------------------ Final Answer Refinement ------------------}
    $A^* \gets \LLM(a^{best}, P_f)$ \tcp{Extract clean final answer sentence}

    \KwRet{$A^*$}
    
\end{algorithm}
